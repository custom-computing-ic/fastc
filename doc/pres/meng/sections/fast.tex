  void kernel_Sum(float* in, float* sum) {
    float prev = CURRENT_CYCLE > 0 ? in[-1] : 0;
    float next = CURRENT_CYCLE < MAX_CYCLE ? in[1] : 0;
    sum[0]  = prev + in[0] + next;
  }
  \end{lstlisting}
